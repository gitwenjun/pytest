## 目录结构
    allure_report           --生成报告
    allure_temps            --报告临时目录
    common                  --公共包
    config                  --配置文件
    data                    --数据
    logs                    --存放日志
    pag                     --Page类
    page_element            --定位元素
    page_object             --函数方法
    photos                  --存放截图
    send_email              --发送邮件
    testcase_py             --测试用例
    pytest.ini              --pytest配置文件
---
